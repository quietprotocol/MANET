# Restore PL011 (ttyAMA0) to GPIOs 14 & 15, instead of Mini UART (ttyS0).
# Mini UART is disabled by default unless "enable_uart=1" is specified,
#  which changes the core frequency to a fixed value and impacts performance.
# See https://www.raspberrypi.com/documentation/computers/configuration.html#mini-uart-and-cpu-core-frequency
[pi0w]
dtoverlay=disable-bt
[pi3]
dtoverlay=disable-bt
[pi4]
dtoverlay=miniuart-bt
[cm4]
# This overlay is used to allow m.2 wifi cards
dtoverlay=pcie-32bit-dma
# Enable USB ports on carrier cards
dtoverlay=dwc2,dr_mode=host

[all]
# Disable overscan assuming the display supports displaying the full resolution
# If the text shown on the screen disappears off the edge, comment this out
disable_overscan=1

dtparam=i2c_arm=off

# How much memory in MB to assign to the GPU on Pi models having
# 256, 512 or 1024 MB total memory
gpu_mem_256=128
gpu_mem_512=128
gpu_mem_1024=128

# enable uart
uart_enable=1

#enable sdio
dtoverlay=sdio,poll_once=on

#setup sub 50MHz SDIO clock
dtparam=sdio_overclock=42

#enable uart5
dtoverlay=uart5

#enable morse gpio configuration
#dtoverlay=morse-wm1302-cm4.dtbo
dtoverlay=mm610x-spi.dtbo
#adds morse-ps device tree  6.12 fixing
#dtoverlay=morse-ps

#enable spi
dtparam=spi=on

#disable SD activity LED by default
#(we configure it for HaLow later in userspace)
dtparam=act_led_trigger=none

#adds ramoops device tree overlay
dtoverlay=ramoops

# automatically load sensor dtoverlays
camera_auto_detect=1

dtoverlay=sysinfo,board-name="morse,ekh01",model="MorseMicro EKH01"
#dtoverlay=mm610x-spi
#dtoverlay=fixed-morse

# FORCE HARDWARE ALWAYS ON
# GPIO 3 (Power1), 7 (Power2), 17 (Reset) -> Output, Driven High
gpio=3=op,dh
gpio=7=op,dh
gpio=17=op,dh
